## 合并排序的数组

![](./images/052.png)

```python
class Solution:
    def merge(self, A: List[int], m: int, B: List[int], n: int) -> None:
        """
        Do not return anything, modify A in-place instead.
        """
        if n:
            A[m] = B[0]
            point = m
            while(point>0):
                if A[point]<A[point-1]:
                    mid = A[point-1]
                    A[point-1] = A[point]
                    A[point] = mid
                    point -= 1
                else:
                    break
            return self.merge(A,m+1,B[1:],n-1)
        else:
            return A
```

## 稀疏数组搜索

![](./images/55.png)

```python
class Solution:
    def findString(self, words: List[str], s: str) -> int:
        first = 0
        end = len(words)-1
        while(end - first > 1):
            mid = (first+end)//2
            mid2 = mid
            while(words[mid2]==""):
                if mid2 + 1<end:
                    mid2+=1
                    continue
                end = mid
                break
            if words[mid2]==s:
                return mid2
            elif words[mid2]<s:
                first = mid2
            else:
                end = mid
        
        if words[first]==s:
            return first
        elif words[end]==s:
            return end
        else:
            return -1
```
