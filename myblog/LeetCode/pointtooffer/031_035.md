class Solution:
    def verifyPostorder(self, postorder: List[int]) -> bool:
        class ListNode():
            def __init__(self,x) -> None:
                self.val = x
                self.left = None
                self.right = None
            
            def bulid(self,listval):
                for val in listval:
                    self.insert(val)
                    
            def insert(self, val):
                if self.val > val:
                    if self.left:
                        self.left.insert(val)
                    else:
                        self.left = ListNode(val)
                elif self.val < val:
                    if self.right:
                        self.right.insert(val)
                    else:
                        self.right = ListNode(val)
